<program-list> -> <function> <program-list> 
           | ;;

<function> -> <Type> <Name> ( <variable-list> ) { <code-list> } ;;

<variable-list> -> <Type> <Name> , <variable-list>
                 | <Type> <Name> 
				 | ;;

<code-list> -> <operator> <code-list> 
              |  ;;

<operator> -> <conditional-operator> 
            | <decalaration-operator> 
			| <while-cycle>
			| <expression> ; ;;

<conditional-operator> -> if ( <expression> ) { <code-list> } else { <code-list> } ;;

<decalaration-operator> -> <Type> <Name> ;;

<while-cycle> -> while ( <expression> ) { <code-list> } ;;

<expression> -> ( <expression> <Operator> <expression> ) 
              | ( <Operator> <expression> ) 
			  | <Name> ( <argument-list> )
			  | <Name>  ;;
			  
<argument-list> -> <Name> , <argument-list>
                 | <Name>
				 | ;;
